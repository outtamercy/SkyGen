unit ExportPluginData;

uses
  SysUtils, Classes, JSON, Mutex, xEditAPI;

// Helper function to get command-line arguments (like -o:outputPath)
function GetCommandLineArg(const ArgName: string): string;
var
  i: Integer;
begin
  Result := '';
  for i := 0 to ParamCount - 1 do
  begin
    if StartsStr(ArgName, ParamStr(i)) then
    begin
      Result := Copy(ParamStr(i), Length(ArgName) + 1, MaxInt);
      Exit;
    end;
  end;
end;

// Simple logging function for xEdit's message log
procedure Log(const s: string);
begin
  AddMessage(s, mtStatus);
end;

// Main processing function for the xEdit script
function Process: Integer;
var
  OutputPath, SourceModFileName: string;
  OutputJSON: TJSONObject; // Main JSON object for output
  BaseObjectsNode: TJSONObject; // Dictionary for all base objects
  SourceModBaseObjectsNode: TJSONArray; // Array for source mod's base objects
  i, j: Integer;
  CurFile: IInterface; // Represents a loaded plugin file
  CurRecord: IInterface; // Represents a record within a plugin
  RecordSignature, RecordFormID, RecordEDID, OriginModName: string;
  RecordData: TJSONObject; // Temporary object for current record's data
  RelevantSignatures: TStringList; // List of record types (categories) to export
begin
  Result := 1; // Default to failure

  // Get output path and source mod filename from command-line arguments
  OutputPath := GetCommandLineArg('-o:');
  SourceModFileName := GetCommandLineArg('-s:'); // New argument for source mod

  if OutputPath = '' then
  begin
    Log('Error: Output path not specified. Use -o:<path>');
    Exit;
  end;

  Log('Exporting plugin data to: ' + OutputPath);
  if SourceModFileName <> '' then
    Log('Filtering source mod base objects for: ' + SourceModFileName);

  OutputJSON := TJSONObject.Create;
  BaseObjectsNode := TJSONObject.Create;
  SourceModBaseObjectsNode := TJSONArray.Create;

  // Define the record signatures (categories) we are interested in
  RelevantSignatures := TStringList.Create;
  RelevantSignatures.Add('FURN'); // Furniture
  RelevantSignatures.Add('STAT'); // Static Objects
  RelevantSignatures.Add('WEAP'); // Weapons
  RelevantSignatures.Add('ARMO'); // Armor
  RelevantSignatures.Add('CONT'); // Containers
  RelevantSignatures.Add('LIGH'); // Lights
  RelevantSignatures.Add('DOOR'); // Doors
  RelevantSignatures.Add('ACTI'); // Activators
  RelevantSignatures.Add('TREE'); // Trees
  // Add any other relevant base object signatures here if needed

  try
    // Iterate through all loaded files (plugins) in xEdit
    for i := 0 to FileCount - 1 do
    begin
      CurFile := FileByIndex(i);
      OriginModName := GetFileName(CurFile);
      Log('Processing file: ' + OriginModName);

      // Iterate through all top-level records in the current file
      for j := 0 to ElementCount(CurFile) - 1 do
      begin
        CurRecord := ElementByIndex(CurFile, j);

        // Skip master records and group records (we only care about actual game objects)
        if not IsMaster(CurRecord) and not IsGroup(CurRecord) then
        begin
          RecordSignature := Signature(CurRecord); // e.g., 'FURN', 'STAT'

          // Check if the current record's signature is one we want to export
          if RelevantSignatures.IndexOf(RecordSignature) <> -1 then
          begin
            RecordFormID := FormID(CurRecord); // Get the FormID (unique identifier)
            RecordEDID := GetEditorID(CurRecord); // Get the Editor ID (human-readable ID)

            // Create a JSON object for this specific record's data
            RecordData := TJSONObject.Create;
            RecordData.AddPair('formId', RecordFormID);
            RecordData.AddPair('EDID', RecordEDID);
            RecordData.AddPair('category', RecordSignature);
            RecordData.AddPair('originMod', OriginModName);

            // Add this record's data to the general 'baseObjects' dictionary
            // BaseObjectsNode takes ownership of RecordData
            BaseObjectsNode.AddPair(RecordFormID, RecordData);

            // If a source mod filename was provided and this record originates from it,
            // add a *clone* of the record's data to the 'sourceModBaseObjects' array.
            // Cloning is necessary because TJSONObject ownership is transferred.
            if (SourceModFileName <> '') and (CompareText(OriginModName, SourceModFileName) = 0) then
            begin
              SourceModBaseObjectsNode.AddElement(RecordData.Clone as TJSONObject);
            end;
          end;
        end;
      end;
    end;

    // Add the collected 'baseObjects' and 'sourceModBaseObjects' to the main output JSON
    OutputJSON.AddPair('baseObjects', BaseObjectsNode); // OutputJSON takes ownership of BaseObjectsNode
    OutputJSON.AddPair('sourceModBaseObjects', SourceModBaseObjectsNode); // OutputJSON takes ownership of SourceModBaseObjectsNode

    // Save the final JSON object to the specified output file
    OutputJSON.SaveToStream(TFileStream.Create(OutputPath, fmCreate));
    Log('Data successfully exported to: ' + OutputPath);
    Result := 0; // Indicate success
  except
    on E: Exception do // Catch any errors during the process
    begin
      Log('Error during export: ' + E.Message);
      Result := 1;
    end;
  end;

  // Free allocated memory
  RelevantSignatures.Free;
  OutputJSON.Free; // This will correctly free BaseObjectsNode and SourceModBaseObjectsNode as well
end;

end.
